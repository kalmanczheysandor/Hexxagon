5   /**
6    * Describes generalised methods of all possible type of players in controller layer.
7    * Artificial player and human player should implement these methods.
8    *
9    * @author Sandor Kalmanchey
10   * @version 2.0
11   * @since 1.0
12   */
13  public interface IPlayer extends Serializable {
14  
15      /**
16       * Retrieves the colour of player.
17       *
18       * @return The colour code of player in html style like <code>#cc0000</code>
19       */
20      public String getPlayerColor();
21  
22      /**
23       * Retrieves the unique index of the player.
24       *
25       * @return An higher than 0 integer value.
26       */
27      public byte getPlayerIndex();
28  
29      /**
30       * Sets the unique index of the player.
31       *
32       * @param playerIndex Value should be higher than 0.
33       */
34      public void setPlayerIndex(byte playerIndex);
35  
36      /**
37       * Tells whether the player is defeated.
38       *
39       * @return True if player is defeated otherwise false.
40       */
41      public boolean isDefeated();
42  
43      /**
44       * Set the defeated status of player.
45       *
46       * @param isDefeated True if player is defeated otherwise false.
47       */
48      public void setIsDefeated(Boolean isDefeated);
49  
50      /**
51       * Tells whether the player is an artificial player.
52       *
53       * @return True if the player is artificial otherwise false.
54       */
55      public boolean isAutoPlayer();
56  }
