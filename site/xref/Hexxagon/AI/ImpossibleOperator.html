27      /**
28       * Generates all available steps represented by this type of operator.
29       *
30       * @param boardState         The specified state on which all apliciable operator is generated.
31       * @param offeredPlayerIndex Index of supported player.
32       *
33       * @return A list operator in all possible variation.
34       *
35       * @throws TError Thrown if an unrecoverable error was occurred.
36       */
37      public static ArrayList<IOperator> generateAvailableOperatorsOnState(IBoardState boardState, byte offeredPlayerIndex) throws TError {
38          // Input checkings
39          if(boardState == null) {
40              throw new TError("Input parameter should not be null!");
41          }
42  
43          if(!isValidBoardState(boardState)) {
44              throw new TError("Invalid board was given!");
45          }
46  
47          if(boardState.getActualPlayerIndex() != offeredPlayerIndex && boardState.getEnemyPlayerIndex() != offeredPlayerIndex) {
48              throw new TError("Invalid player index was given");
49          }
50  
51          // Adding a not appliciable
52          ArrayList<IOperator> output = new ArrayList<>();
53          IOperator opr = new ImpossibleOperator(0, 0, 0, 0);
54          output.add(opr);
55  
56          return output;
57      }
