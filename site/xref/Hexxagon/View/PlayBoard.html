279      * Retrieves the pit object found at specified coordinates.

281      * @param columnIndex Column index of specified coordinate.
282      * @param rowIndex    Row index of specified coordinate.
283      *
284      * @return An instance of {@code BoardPit} class.
285      *
286      * @throws TError Thrown if an unrecoverable error was occurred.
287      */
288     private BoardPit getPit(int columnIndex, int rowIndex) throws TError {
289         // Input checkings
290         BoardPit pit = new BoardPit(this, 0, 0, columnIndex, rowIndex);
291         if(!this.pitList.contains(pit)) {
292             throw new TError("No data was found at given parameters [" + columnIndex + "," + rowIndex + "]!");
293         }
294 
295         // Find data
296         int indx = this.pitList.indexOf(pit);
297         return this.pitList.get(indx);
298     }

375     /**
376      * {@inheritDoc}
377      */
378     @Override
379     public void animateAsAttackZone(int baseColumnIndex, int baseRowIndex) throws TError {
380         // Input checkings
381         if(!this.getDataInstance().isCellExists(baseColumnIndex, baseRowIndex)) {
382             throw new TError("The requested coordinate[" + baseColumnIndex + "," + baseRowIndex + "] is out of range!");
383         }
384         if(!this.isPitExists(baseColumnIndex, baseRowIndex)) {
385             throw new TError("The requested pit[" + baseColumnIndex + "," + baseRowIndex + "] is not found!");
386         }
387 
388         voiceShowSpreadArea.play();
389         // Remove possible recent animation
390         this.removeAllAttackZone();
391 
392         // Animate base cell of attack zone
393         this.animateAsAttackZoneOrigo(baseColumnIndex, baseRowIndex);
394 
395         // Animate cells of attack zone
396         for(Coordinate relativeCoordinate : relativeCoordinatesOfAttackZone) {
397             int zoneColumnIndex = baseColumnIndex + relativeCoordinate.getX();
398             int zoneRowIndex = baseRowIndex + relativeCoordinate.getY();
399 
400             if(this.isPitExists(zoneColumnIndex, zoneRowIndex) && this.getDataInstance().isCellExists(zoneColumnIndex, zoneRowIndex)) {
401 
402                 BoardPit pit = this.getPit(zoneColumnIndex, zoneRowIndex);
403                 IField field = this.getDataInstance().getCellAsField(zoneColumnIndex, zoneRowIndex);
404                 if((pit.isInactive() == false) && (field.getPlayerIndex() == 0)) {
405                     pit.animateAsAttackZoneCell();
406                 }
407 
408             }
409         }
410     }

807         /**
808          * ????????
809          *
810          * @throws TError Thrown if an unrecoverable error was occurred.
811          */
812         private void checkState() throws TError {
813             IField Field = this.parent.getDataInstance().getCellAsField(this.columnIndex, this.rowIndex);
814 
815             //---Note:
816             if(Field.isInactive())//Note: Ha auz adatmez inaktv cellt jell
817             {
818                 this.setFill(null);
819                 this.setStrokeWidth(1);
820             }
821             else {
822                 //---Note: Felhasznl sznnek lltsa
823                 if(Field.getPlayerIndex() != 0)//Note: Ha az adott cella egy jtkos ltal foglalva van
824                 {
825                     IPlayer Player = this.parent.getDataInstance().getPlayer(Field.getPlayerIndex());
826 
827                     String ColorString = Player.getPlayerColor();
828 
829                     if(ColorString == null) {
830                         ColorString = "#000000";
831                     }
832 
833                     RadialGradient PlayerColorFill = new RadialGradient(0, 0, 0.3, 0.3, 1, true, CycleMethod.NO_CYCLE, new Stop[]{
834                         new Stop(0, Color.web(ColorString)),
835                         new Stop(1, Color.BLACK)
836                     });
837 
838                     this.setFill(PlayerColorFill);
839 
840                 }
841                 if(Field.getPlayerIndex() == 0)//Note: Ha az adott cellt nem foglalja jtkos
842                 {
843                     this.setFill(BoardPit.emptyBackgroundFill);
844                 }
845                 //---
846 
847             }
848             //---
849 
850         }
