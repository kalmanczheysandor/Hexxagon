17  /**
18   * Object instance of this class is responsible for the initialisation and draw of graphical component displayed on play page.
19   * Contains the implementation of application page/view component specified by {@code ISettingsPage} interface.
20   * The class extends the {@code JFXPage} javaFX specific abstract page class.
21   *
22   * @author Sandor Kalmanchey
23   * @version 1.0
24   * @since 1.0
25   */
26  public class PlayPage extends JFXPage implements IPlayPage {
27  
28      /**
29       * The logger.
30       */
31      private static final Logger logger = LoggerFactory.getLogger(PlayPage.class);
32  
33      /**
34       * Object of play board graphical component of Hexxagon module.
35       */
36      private IPlayBoard playBoard;
37      /**
38       * Object of info board graphical component of Hexxagon module.
39       */
40      private IInfoBoard infoBoard;
41      
42       /**
43       * Object of faceBoard graphical component of Hexxagon module.
44       */
45      private IFaceBoard faceBoard;
46      
47      /**
48       * Graphical node on which placed the graphical components of Hexxahon module.
49       */
50      private AnchorPane gameNode;
51      /**
52       * Object of back button graphical component.
53       */
54      private Button backButton;
55  
56      /**
57       * Creates a {@code PlayPage} instance.
58       *
59       * @param parentNode Outside graphical root node.
60       */
61      public PlayPage(Object parentNode) {
62          super(parentNode);
63      }
64  
65      /**
66       * {@inheritDoc}
67       */
68      @Override
69      protected void initComponent() throws TError{
70  
71          // Inatialise a component
72          this.backButton = new Button("Back");
73          this.backButton.setOnMouseClicked((event) -> {
74  
75              try {
76                  this.getContainer().stopGame();
77              }
78              catch(TError ex) {
79                   logger.error(ex.toString());
80              }
81              catch(Exception ex) {
82                   logger.error(ex.toString());
83              }
84          });
85  
86          // The graphical root for hexagon boards
87          this.gameNode = new AnchorPane();
88          
89          Pane playBoardNode= new Pane();
90          Pane infoBoardNode= new Pane();
91          Pane faceBoardNode= new Pane();
92  
93          // Inatialise a component
94          playBoard = new PlayBoard(playBoardNode);
95  
96          // Inatialise a component
97          infoBoard = new InfoBoard(infoBoardNode);
98          
99          // Inatialise a component
100         faceBoard = new FaceBoard(faceBoardNode);
101         
102         
103         
104         this.gameNode.getChildren().add(faceBoardNode);
105         this.gameNode.setRightAnchor(faceBoardNode, Double.valueOf(-20));
106         this.gameNode.setTopAnchor(faceBoardNode, Double.valueOf(-20));
107       
108         this.gameNode.getChildren().add(playBoardNode);
109         this.gameNode.setTopAnchor(playBoardNode, Double.valueOf(0));
110         this.gameNode.setTopAnchor(playBoardNode, Double.valueOf(0));
111         this.gameNode.getChildren().add(infoBoardNode);
112             
113             
114        
115     }
116 
117     /**
118      * {@inheritDoc}
119      */
120     @Override
121     protected Object drawContent() throws TError{
122 
123         AnchorPane root = new AnchorPane();
124        
125         root.getChildren().add(this.backButton);
126         root.getChildren().add(this.gameNode);
127         
128         AnchorPane.setLeftAnchor(this.gameNode, Double.valueOf(0));
129         AnchorPane.setRightAnchor(this.gameNode, Double.valueOf(0));
130         AnchorPane.setTopAnchor(this.gameNode, Double.valueOf(0));
131         AnchorPane.setBottomAnchor(this.gameNode, Double.valueOf(40));
132         
133         AnchorPane.setLeftAnchor(this.backButton, Double.valueOf(20));
134         AnchorPane.setBottomAnchor(this.backButton, Double.valueOf(20));
135 
136         // Set css style
137         root.setId("RootNode");
138         return root;
139     }
140 
141     /**
142      * {@inheritDoc}
143      */
144     @Override
145     public void onAfterDraw() {
146 
147     }
148

150      * {@inheritDoc}
151      */
152     @Override
153     public IPlayBoard getPlayBoard() {
154         return playBoard;
155     }
156 
157     /**
158      * {@inheritDoc}
159      */
160     @Override
161     public IInfoBoard getInfoBoard() {
162         return infoBoard;
163     }
164 
165     
166     /**
167      * {@inheritDoc}
168      */
169     @Override
170     public IFaceBoard getFaceBoard() {
171         return faceBoard;
172     }
173     
174     
175     /**
176      * {@inheritDoc}
177      */
178     @Override
179     public void modelPropertyChange(PropertyChangeEvent evt) throws TError {
180         logger.trace("Property change triggered at PlayPage");
181     }

183 }
