7   
8   import Main.Main;
9   import Support.TError;
10  import java.beans.PropertyChangeEvent;
11  import javafx.scene.control.Button;
12  import javafx.scene.image.Image;
13  import javafx.scene.image.ImageView;
14  import javafx.scene.layout.AnchorPane;
15  import javafx.scene.text.Text;
16  import javafx.scene.text.TextAlignment;
17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;

20  /**
21   *
22   * @author ALX
23   */
24  public class InfoPage extends JFXPage implements IInfoPage {
25  
26      /**
27       * The logger.
28       */
29      private static final Logger logger = LoggerFactory.getLogger(IntroPage.class);
30  
31      /**
32       * Object of towards button graphical component.
33       */
34      private Button backButton;
35  
36      private static final Image iconTitle = new Image(InfoPage.class.getResourceAsStream(Main.folderImage + "/title_left.gif"));
37      private ImageView imageTitle;
38      private Text authorText;
39      private Text storyBlock;
40      /**
41       * Creates a {@code InfoPage} instance.
42       *
43       * @param parentNode Outside graphical root node.
44       */
45      public InfoPage(Object parentNode) {
46          super(parentNode);
47      }
48  
49      @Override
50      protected void initComponent() throws TError {
51          // Component initialisation
52          this.backButton = new Button("Back");
53          this.backButton.setOnMouseClicked((event) -> {
54  
55              try {
56                  this.getContainer().showIntroPage();
57              }
58              catch(TError ex) {
59                  logger.error(ex.toString());
60              }
61              catch(Exception ex) {
62                  logger.error(ex.toString());
63              }
64  
65          });
66          
67          // Component initialisation
68          imageTitle = new ImageView(iconTitle);
69          imageTitle.setScaleX(1);
70          imageTitle.setScaleY(0.9);
71          
72         // Component initialisation
73          authorText = new Text(0, 0, ""
74          + "Author: Sandor Kalmanchey, Pti Bsc. \n"
75          + "E-mail: kalmanczheysandor@gmail.com\n"
76          + "Neptun: Z2J914");
77          authorText.setId("Text");
78          authorText.setTextAlignment(TextAlignment.LEFT);
79          
80          // Component initialisation
81          storyBlock = new Text(0, 0, ""
82          + "This application was made as accomplishment for exam requirements in the following subjects:\n"
83          + "- ILDV444 Mesterséges intelligencia alkalmazások \n"
84          + "- ILDK311 Programozási környezetek\n"
85          + "at University of Debrecen Faculty of Informatics.\n"
86          + "After the first release in 2015 the application was redesigned in 2019.\n");
87          storyBlock.setId("Text");
88          
89      }
90  
91      @Override
92      protected Object drawContent() throws TError {
93          AnchorPane root = new AnchorPane();
94          root.setId("RootNode");
95  
96          // Attaching components
97          root.getChildren().add(backButton);
98          root.getChildren().add(imageTitle);
99          root.getChildren().add(authorText);
100         root.getChildren().add(storyBlock);
101         
102 
103         // Anchoring of back button
104         AnchorPane.setLeftAnchor(this.backButton, Double.valueOf(20));
105         AnchorPane.setBottomAnchor(this.backButton, Double.valueOf(20));
106 
107         // Anchoring the title
108         AnchorPane.setTopAnchor(imageTitle, Double.valueOf(-20));
109         AnchorPane.setLeftAnchor(imageTitle, Double.valueOf(20));
110         
111         // Anchoring the author text
112         AnchorPane.setRightAnchor(authorText, Double.valueOf(50));
113         AnchorPane.setBottomAnchor(authorText, Double.valueOf(200));
114 
115         // Anchoring the story block text
116         AnchorPane.setLeftAnchor(storyBlock, Double.valueOf(100));
117         AnchorPane.setTopAnchor(storyBlock, Double.valueOf(200));
118         return root;
119     }
120 
121     @Override
122     public void modelPropertyChange(PropertyChangeEvent evt) throws TError {
123 
124     }
125 
126     @Override
127     public void onAfterDraw() throws TError {
128 
129     }

131 }
