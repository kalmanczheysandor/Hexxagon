1   package Support.mvc;
2   
3   import Support.TError;
4   import java.beans.PropertyChangeEvent;
5   
6   /**
7    * Contains generalised methods of view layer of MVC design pattern.
8    *
9    * @author Sandor Kalmanchey
10   * @version 1.0
11   * @since 1.0
12   */
13  public interface IView {
14  
15      /**
16       * Tells whether a controller component is attached.
17       *
18       * @return True if controller component attached to the view component and false otherwise.
19       */
20      public boolean isControllerAttached();
21  
22      /**
23       * Receives the property change event occurred in the model component. The event is delegated by the controller component.
24       *
25       * @param evt Describes the details of property change occurred in the model component.
26       *
27       * @throws TError Thrown if an unrecoverable error was occurred.
28       */
29      public void modelPropertyChange(final PropertyChangeEvent evt) throws TError;
30  
31      /**
32       * Attaches a controller object to the view component. Only one is available at same time.
33       *
34       * @param controller The controller object to attach.
35       *
36       * @throws TError Thrown if an unrecoverable error was occurred.
37       */
38      public void addController(IController controller) throws TError;
39  
40      /**
41       * Detaches the previously added controller component.
42       */
43      public void removeController();
44  
45      /**
46       * Retrieves the reference of the previously added controller component.
47       *
48       * @return Reference to the added controller component. Null if no component is added yet.
49       */
50      public IController getController();
51  }
