1   package Hexxagon.Controller;
2   
3   import Hexxagon.Model.IGameData;
4   import Hexxagon.View.IFaceBoard;
5   import Hexxagon.View.IInfoBoard;
6   import Hexxagon.View.IPlayBoard;
7   import Support.TError;
8   
9   /**
10   * Contains generalised methods of controller layer of MVC design pattern.
11   *
12   * @author Sandor Kalmanchey
13   * @version 2.0
14   * @since 1.0
15   */
16  public interface IGameController {
17  
18      /**
19       * Starts the game play.
20       *
21       * @throws TError Thrown if an unrecoverable error was occurred.
22       */
23      public void startGame() throws TError;
24  
25      /**
26       * Interrupts the game.
27       * If invoked in inappropriate game state than {@code TError } is thrown.
28       *
29       * @throws TError Thrown if an unrecoverable error was occurred.
30       */
31      public void pauseGame() throws TError;
32  
33      /**
34       * Allows to continue the game.
35       * If invoked in inappropriate game state than {@code TError } is thrown.
36       *
37       * @throws TError Thrown if an unrecoverable error was occurred.
38       */
39      public void continueGame() throws TError;
40  
41      /**
42       * Stops the game.
43       * If invoked in inappropriate game state than {@code TError } is thrown.
44       *
45       * @throws TError Thrown if an unrecoverable error was occurred.
46       */
47      public void stopGame() throws TError;
48  
49      /**
50       * Tells whether the game is active.
51       * If a game is not active than no human or artificial player can invoke events.
52       * @return True if the game is running otherwise false.
53       * @throws TError Thrown if an unrecoverable error was occurred.
54       */
55      
56      public boolean isGameRunning() throws TError;
57      
58      
59      
60      /**
61       * Informs the controller module about a step from a position to another one.
62       * This method is used only by artificial player.
63       *
64       * @param fromColumnIndex   Column index of the cell from where the step is taken. Index should be a valid one otherwise {@code TError} is thrown.
65       * @param fromRowIndex      Row index of the cell from where the step is taken. Index should be a valid one otherwise {@code TError} is thrown.
66       * @param targetColumnIndex Column index of the cell to where the step is taken. Index should be a valid one otherwise {@code TError} is thrown.
67       * @param targetRowIndex    Row index of the cell to where the step is taken. Index should be a valid one otherwise {@code TError} is thrown.
68       * @param playerIndex       The unique index of player. Index should be a valid one otherwise {@code TError} is thrown.
69       *
70       * @throws TError Thrown if an unrecoverable error was occurred.
71       */
72      public void autoPlayerCellStep(int fromColumnIndex, int fromRowIndex, int targetColumnIndex, int targetRowIndex, byte playerIndex) throws TError;
73  
74      /**
75       * Informs the controller module about that a cell was clicked by a human player.
76       * This method is used only by the human player.
77       *
78       * @param columnIndex Column index of the cell clicked. Index should be a valid one otherwise {@code TError} is thrown.
79       * @param rowIndex    Row index of the cell clicked. Index should be a valid one otherwise {@code TError} is thrown.
80       *
81       * @throws TError Thrown if an unrecoverable error was occurred.
82       */
83      public void cellClicked(int columnIndex, int rowIndex) throws TError;
84  
85      /**
86       * Sets the data component of the game.
87       * Only one data component is allowed to being attached to controller.
88       *
89       * @param gameData An object instance of class implementing {@code IGameData} interface. Parameter should not be null otherwise {@code TError} is thrown.
90       *
91       * @throws TError Thrown if an unrecoverable error was occurred.
92       */
93      public void setGameData(IGameData gameData) throws TError;
94  
95      /**
96       * Adds a play board visual component to the controller.
97       * Play board is the visual interface on which the balls are placed.
98       * More than one play board component is allowed to being attached to controller.
99       *
100      * @param playBoard An object instance of class implementing {@code IPlayBoard} interface. Parameter should not be null otherwise {@code TError} is thrown.
101      *
102      * @throws TError Thrown if an unrecoverable error was occurred.
103      */
104     public void addPlayBoard(IPlayBoard playBoard) throws TError;
105 
106     /**
107      * Adds a info board visual component to the controller.
108      * Info board is the visual interface on which the actual points of players are presented.
109      * More than one info board component is allowed to being attached to controller.
110      *
111      * @param infoBoard An object instance of class implementing {@code IInfoBoard} interface. Parameter should not be null otherwise {@code TError} is thrown.
112      *
113      * @throws TError Thrown if an unrecoverable error was occurred.
114      */
115     public void addInfoBoard(IInfoBoard infoBoard) throws TError;
116     
117     
118     /**
119      * Adds a face board visual component to the controller.
120      * Face board is the visual interface on which the artificial player is presented.
121      * More than one info board component is allowed to being attached to controller.
122      *
123      * @param faceBoard An object instance of class implementing {@code IFaceBoard} interface. Parameter should not be null otherwise {@code TError} is thrown.
124      *
125      * @throws TError Thrown if an unrecoverable error was occurred.
126      */
127     public void addFaceBoard(IFaceBoard faceBoard) throws TError;
128 }
